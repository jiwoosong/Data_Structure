## 7. Polymorphism
### - Example 1
```C++
class parent {
public:
	virtual void house(void) { cout << "parent's house" << endl; }
};

class child : public parent
{
public:
	void house(void) { cout << "child's house" << endl; }
};

int main(void) 
{
	parent* pp = new parent();
	child* cc = new child();
	parent* pc = new child();

	pp->house();
	cc->house();
	pc->house();
}
```
